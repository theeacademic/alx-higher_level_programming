import unittest
from your_module import add_integer  # replace 'your_module' with the name of your module

class TestAddInteger(unittest.TestCase):
    def test_add_integer(self):
        self.assertEqual(add_integer(1, 2), 3)
        self.assertEqual(add_integer(-1, 2), 1)
        self.assertEqual(add_integer(1.5, 2.5), 3)
        self.assertRaises(TypeError, add_integer, '1', 2)
        self.assertRaises(TypeError, add_integer, 1, '2')

if __name__ == '__main__':
    unittest.main()
